#!/data/data/com.termux/files/usr/bin/bash

G='\033[0;32m'; B='\033[1;34m'; N='\033[0m'

echo -e "\n${B}⚡ HamidScanner ULTRA — 2000 Parallel Cloudflare Scan${N}\n"

# بهترین رنج‌های Cloudflare
ranges=(
104.16 104.17 104.18 104.19 104.20 104.21 104.22 104.23 104.24 104.25 104.26
172.64 172.65 172.66 172.67 172.68 172.69
188.114
)

results="results.tmp"
rm -f $results

total=2000
threads=50
running=0

scan_ip() {
    ip=$1
    ping_output=$(ping -c1 -W1 $ip 2>/dev/null)
    ms=$(echo "$ping_output" | grep 'time=' | awk -F'time=' '{print $2}' | awk '{print $1}')
    [[ -n "$ms" ]] && echo "$ip,$ms" >> $results
}

for ((i=1; i<=total; i++)); do
    prefix=${ranges[$RANDOM % ${#ranges[@]}]}
    ip="$prefix.$((RANDOM % 256)).$((RANDOM % 256))"

    echo -ne "${G}[$i/$total]${N} Testing $ip\r"

    scan_ip $ip &

    ((running++))
    if (( running >= threads )); then
        wait
        running=0
    fi
done

wait

echo -e "\n\n${B}Sorting results...${N}\n"

echo -e "${B}🔥 Top 10 Fastest Cloudflare IPs:${N}"
sort -t',' -k2,2n $results | head -n 10 | nl

echo -e "\n${G}✔ Done — HamidScanner ULTRA${N}\n"
