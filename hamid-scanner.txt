#!/bin/bash

echo "Generating Cloudflare IP list..."
rm -f ips.txt

# Best Cloudflare prefixes
prefixes=(
104.16
104.17
104.18
104.19
104.20
104.21
104.22
104.23
104.24
104.25
104.26
172.64
172.65
172.66
172.67
172.68
172.69
188.114
)

# Generate IPs without prips
for p in "${prefixes[@]}"; do
    for i in $(seq 0 255); do
        for j in $(seq 1 254); do
            echo "$p.$i.$j" >> ips.txt
        done
    done
done

echo "Total IPs generated: $(wc -l < ips.txt)"
echo "Starting scan..."

rm -f result.csv
echo "IP,RTT" > result.csv

while read ip; do
    rtt=$(ping -c1 -W1 $ip 2>/dev/null | awk -F'=' '/time=/{print $4}' | cut -d' ' -f1)
    if [ ! -z "$rtt" ]; then
        echo "$ip,$rtt" >> result.csv
        echo "OK  $ip  $rtt ms"
    fi
done < ips.txt

echo
echo "Scan finished."
echo "Top 20 fastest IPs:"
sort -t',' -k2,2n result.csv | head -n 20
