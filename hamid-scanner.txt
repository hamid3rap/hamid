#!/bin/bash

echo "Generating Cloudflare IP list..."
rm -f ips.txt

# Best Cloudflare ranges
ranges=(
104.16.0.0/20
104.17.0.0/20
104.18.0.0/20
104.19.0.0/20
104.20.0.0/20
104.21.0.0/20
104.22.0.0/20
104.23.0.0/20
104.24.0.0/20
104.25.0.0/20
104.26.0.0/20
172.64.0.0/16
172.65.0.0/16
172.66.0.0/16
172.67.0.0/16
172.68.0.0/16
172.69.0.0/16
188.114.96.0/20
188.114.112.0/20
)

# Convert ranges to IP list
for r in "${ranges[@]}"; do
    prips $r >> ips.txt
done

echo "Total IPs generated: $(wc -l < ips.txt)"
echo "Starting scan..."

rm -f result.csv
echo "IP,RTT" > result.csv

while read ip; do
    rtt=$(ping -c1 -W1 $ip 2>/dev/null | awk -F'=' '/time=/{print $4}' | cut -d' ' -f1)
    if [ ! -z "$rtt" ]; then
        echo "$ip,$rtt" >> result.csv
        echo "OK  $ip  $rtt ms"
    fi
done < ips.txt

echo
echo "Scan finished."
echo "Top 20 fastest IPs:"
sort -t',' -k2,2n result.csv | head -n 20
